## 📝 计算公式

### 1. 生命值计算

```typescript
// 基础生命 + 体力加成
maxHP = baseHP + (stamina * hpPerStamina)

// 示例：1 级角色
// baseHP = 100
// hpPerStamina = 10
// stamina = 5
// maxHP = 100 + (5 * 10) = 150
```

### 2. 魔法值计算

```typescript
// 基础魔法 + 智力加成
maxMP = baseMP + (intelligence * mpPerInt)

// 示例：1 级角色
// baseMP = 50
// mpPerInt = 8
// intelligence = 5
// maxMP = 50 + (5 * 8) = 90
```

### 3. 物理攻击计算

```typescript
// 力量主导的物理攻击
physicalAttack = baseAttack + (strength * attackPerStr)

// 示例：1 级角色
// baseAttack = 10
// attackPerStr = 3
// strength = 5
// physicalAttack = 10 + (5 * 3) = 25
```

### 4. 法术攻击计算

```typescript
// 智力主导的法术攻击
magicAttack = baseMagic + (intelligence * magicPerInt)

// 示例：1 级角色
// baseMagic = 15
// magicPerInt = 4
// intelligence = 5
// magicAttack = 15 + (5 * 4) = 35
```

### 5. 防御计算

```typescript
// 体力主导的防御
defense = baseDefense + (stamina * defensePerStamina)

// 示例：1 级角色
// baseDefense = 5
// defensePerStamina = 2
// stamina = 5
// defense = 5 + (5 * 2) = 15
```

## 📊 完整实现代码

```typescript
// src/types/character.ts

interface CharacterAttributes {
  strength: number;     // 力量
  agility: number;      // 敏捷
  intelligence: number; // 智力
  stamina: number;      // 体力
}

interface CharacterStats {
  maxHP: number;
  maxMP: number;
  physicalAttack: number;
  magicAttack: number;
  defense: number;
  speed: number;
}

// src/utils/attributeCalculator.ts

export function calculateStats(
  attributes: CharacterAttributes,
  level: number = 1
): CharacterStats {
  // 基础值（1 级）
  const baseHP = 100;
  const baseMP = 50;
  const baseAttack = 10;
  const baseMagic = 15;
  const baseDefense = 5;
  
  // 成长系数
  const hpPerStamina = 10;
  const mpPerInt = 8;
  const attackPerStr = 3;
  const magicPerInt = 4;
  const defensePerStamina = 2;
  
  return {
    maxHP: baseHP + (attributes.stamina * hpPerStamina),
    maxMP: baseMP + (attributes.intelligence * mpPerInt),
    physicalAttack: baseAttack + (attributes.strength * attackPerStr),
    magicAttack: baseMagic + (attributes.intelligence * magicPerInt),
    defense: baseDefense + (attributes.stamina * defensePerStamina),
    speed: attributes.agility * 0.5,
  };
}
```
