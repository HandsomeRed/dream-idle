## 🔧 架构设计

### 1. 类型定义

```typescript
// src/types/attributes.ts

/** 基础属性 */
export interface BaseAttributes {
  strength: number;     // 力量
  agility: number;      // 敏捷
  intelligence: number; // 智力
  stamina: number;      // 体力
}

/** 战斗属性 */
export interface CombatStats {
  maxHP: number;        // 最大生命
  maxMP: number;        // 最大魔法
  physicalAttack: number; // 物理攻击
  magicAttack: number;    // 法术攻击
  defense: number;        // 防御
  speed: number;          // 速度
}

/** 完整角色属性 */
export interface CharacterAttributes {
  base: BaseAttributes;     // 基础属性
  combat: CombatStats;      // 战斗属性
  bonus: Partial<BaseAttributes>; // 加成属性
}
```

### 2. 属性计算器

```typescript
// src/utils/attributeCalculator.ts

export function calculateStats(
  attributes: BaseAttributes,
  level: number = 1
): CombatStats {
  const baseHP = 100;
  const baseMP = 50;
  const baseAttack = 10;
  const baseMagic = 15;
  const baseDefense = 5;
  
  const hpPerStamina = 10;
  const mpPerInt = 8;
  const attackPerStr = 3;
  const magicPerInt = 4;
  const defensePerStamina = 2;
  
  return {
    maxHP: baseHP + (attributes.stamina * hpPerStamina),
    maxMP: baseMP + (attributes.intelligence * mpPerInt),
    physicalAttack: baseAttack + (attributes.strength * attackPerStr),
    magicAttack: baseMagic + (attributes.intelligence * magicPerInt),
    defense: baseDefense + (attributes.stamina * defensePerStamina),
    speed: attributes.agility * 0.5,
  };
}
```

### 3. 属性点分配

```typescript
// src/utils/attributeAllocator.ts

export function allocatePoints(
  current: BaseAttributes,
  availablePoints: number,
  allocations: Partial<BaseAttributes>
): { attributes: BaseAttributes; remaining: number } {
  const total = Object.values(allocations).reduce((sum, val) => sum + val, 0);
  
  if (total > availablePoints) {
    throw new Error('属性点不足');
  }
  
  const newAttributes = {
    strength: current.strength + (allocations.strength || 0),
    agility: current.agility + (allocations.agility || 0),
    intelligence: current.intelligence + (allocations.intelligence || 0),
    stamina: current.stamina + (allocations.stamina || 0),
  };
  
  return {
    attributes: newAttributes,
    remaining: availablePoints - total,
  };
}
```

## ⚠️ 注意事项

1. **类型安全** - 使用 TypeScript 保证类型正确
2. **纯函数** - 计算函数不修改输入，返回新对象
3. **可测试性** - 逻辑与 UI 分离，便于测试
